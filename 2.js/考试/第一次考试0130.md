### 求和未知参数的和
    ```
    function total(){
        var res=0;
        for(var i=0;i<arguments.length;i++){
            var getItems=Number(arguments[i])
            if(isNaN(getItems)==false){
                res+=getItems
            }
        }
        return res
    }
    total(3,"12","12px")
    ```
### 就既能被 3 又能被 5 除
    ```
    function total(item){
        var res=0
        for(var i=0;i<item;i++){
            if(i%3==0&&i%5==0){
                res+=i
            }
        }
        return res
    }
    total(101)
    ```
